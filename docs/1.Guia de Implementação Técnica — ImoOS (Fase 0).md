### 2.4. `apps/tenants/middleware.py`

```python
from django_tenants.middleware import TenantMainMiddleware
from django.http import Http404
from rest_framework.exceptions import PermissionDenied

class ImoOSTenantMiddleware(TenantMainMiddleware):
    """
    Middleware personalizado com validações adicionais de tenant
    """
    
    def get_tenant(self, model, hostname, request):
        tenant = super().get_tenant(model, hostname, request)
        
        # Validações adicionais
        if not tenant.is_active:
            raise PermissionDenied("Tenant inativo")
        
        # Guardar tenant no request para acesso fácil
        request.tenant = tenant
        
        return tenant
    
    def process_request(self, request):
        response = super().process_request(request)
        
        # Adicionar headers de tenant para debugging
        if hasattr(request, 'tenant'):
            response['X-Tenant-Name'] = request.tenant.name
            response['X-Tenant-Schema'] = request.tenant.schema_name
        
        return response
```

```python
from django.db import models
from simple_history.models import HistoricalRecords
from django_tenants.utils import tenant_context

class Project(models.Model):
    """
    Projeto/Empreendimento (Tenant-specific)
    """
    STATUS_CHOICES = [
        ('planning', 'Planeamento'),
        ('licensing', 'Licenciamento'),
        ('construction', 'Construção'),
        ('completed', 'Concluído'),
        ('cancelled', 'Cancelado'),
    ]
    
    name = models.CharField(max_length=200)
    slug = models.SlugField(max_length=200, unique=True)
    description = models.TextField(blank=True)
    
    location = models.JSONField(default=dict)  # GeoJSON para PostGIS
    address = models.TextField()
    
    status = models.CharField(max_length=20, choices=STATUS_CHOICES, default='planning')
    
    area_total = models.DecimalField(max_digits=10, decimal_places=2, help_text="m²")
    units_count = models.IntegerField(default=0)
    
    budget_total = models.DecimalField(max_digits=15, decimal_places=2, null=True, blank=True)
    
    # Responsáveis
    manager = models.ForeignKey('users.User', on_delete=models.SET_NULL, 
                                 null=True, related_name='managed_projects')
    
    # Datas
    start_date = models.DateField(null=True, blank=True)
    expected_completion = models.DateField(null=True, blank=True)
    completed_at = models.DateField(null=True, blank=True)
    
    # Audit
    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)
    
    history = HistoricalRecords()
    
    class Meta:
        ordering = ['-created_at']
        indexes = [
            models.Index(fields=['status']),
            models.Index(fields=['created_at']),
        ]
    
    def __str__(self):
        return self.name
    
    def save(self, *args, **kwargs):
        # Atualizar units_count automaticamente
        if self.pk:
            self.units_count = self.units.count()
        super().save(*args, **kwargs)

class Unit(models.Model):
    """
    Unidade Imobiliária (Apartamento, Loja, Parqueamento)
    """
    STATUS_CHOICES = [
        ('available', 'Disponível'),
        ('reserved', 'Reservado'),
        ('sold', 'Vendido'),
        ('maintenance', 'Manutenção'),
    ]
    
    TYPE_CHOICES = [
        ('T0', 'T0'),
        ('T1', 'T1'),
        ('T2', 'T2'),
        ('T3', 'T3'),
        ('T4', 'T4'),
        ('shop', 'Loja'),
        ('office', 'Escritório'),
        ('parking', 'Parqueamento'),
        ('storage', 'Arrecadação'),
    ]
    
    project = models.ForeignKey(Project, on_delete=models.CASCADE, related_name='units')
    code = models.CharField(max_length=50)  # Ex: BLK-A-P3-DT
    
    unit_type = models.CharField(max_length=20, choices=TYPE_CHOICES)
    
    area_bruta = models.DecimalField(max_digits=8, decimal_places=2)
    area_util = models.DecimalField(max_digits=8, decimal_places=2, null=True, blank=True)
    
    floor = models.IntegerField(null=True, blank=True)
    building = models.CharField(max_length=50, blank=True)
    block = models.CharField(max_length=50, blank=True)
    
    price = models.DecimalField(max_digits=12, decimal_places=2)
    currency = models.CharField(max_length=3, default='CVE')
    
    status = models.CharField(max_length=20, choices=STATUS_CHOICES, default='available')
    
    # BIM
    bim_guid = models.CharField(max_length=255, blank=True, db_index=True)
    
    # Audit
    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)
    
    history = HistoricalRecords()
    
    class Meta:
        unique_together = ['project', 'code']
        ordering = ['code']
        indexes = [
            models.Index(fields=['status']),
            models.Index(fields=['unit_type']),
            models.Index(fields=['bim_guid']),
        ]
    
    def __str__(self):
        return f"{self.project.name} - {self.code}"
```

```python
import pytest
from django.db import connection
from django_tenants.utils import tenant_context, schema_context
from apps.projects.models import Project
from apps.inventory.models import Unit

@pytest.mark.django_db
class TestTenantIsolation:
    """
    Testes críticos de isolamento entre tenants
    """
    
    def test_tenant_a_cannot_see_tenant_b_projects(self, tenant_a, tenant_b, project_tenant_a, project_tenant_b):
        """Tenant A não deve conseguir ver projetos do Tenant B"""
        
        with tenant_context(tenant_a):
            projects_a = list(Project.objects.all().values_list('id', flat=True))
        
        with tenant_context(tenant_b):
            projects_b = list(Project.objects.all().values_list('id', flat=True))
        
        # Garantir que não há sobreposição de IDs visíveis
        assert set(projects_a).isdisjoint(set(projects_b)), \
            "TENANT ISOLATION FAILED: Tenant A pode ver projetos do Tenant B"
    
    def test_database_schema_separation(self, tenant_a, tenant_b):
        """Garantir que schemas de database estão separados"""
        
        with tenant_context(tenant_a):
            schema_a = connection.schema_name
        
        with tenant_context(tenant_b):
            schema_b = connection.schema_name
        
        assert schema_a != schema_b, "TENANT ISOLATION FAILED: Schemas iguais"
        assert schema_a == tenant_a.schema_name
        assert schema_b == tenant_b.schema_name
    
    def test_unit_creation_isolation(self, tenant_a, tenant_b, project_tenant_a, project_tenant_b):
        """Unidades criadas num tenant não devem aparecer noutro"""
        
        with tenant_context(tenant_a):
            unit_a = Unit.objects.create(
                project=project_tenant_a,
                code='A-001',
                unit_type='T2',
                area_bruta=85.00,
                price=15000000,
            )
            units_a_count = Unit.objects.count()
        
        with tenant_context(tenant_b):
            unit_b = Unit.objects.create(
                project=project_tenant_b,
                code='B-001',
                unit_type='T3',
                area_bruta=120.00,
                price=25000000,
            )
            units_b_count = Unit.objects.count()
        
        # Cada tenant deve ver apenas as suas unidades
        with tenant_context(tenant_a):
            assert Unit.objects.filter(code='B-001').count() == 0, \
                "TENANT ISOLATION FAILED: Tenant A vê unidade do Tenant B"
        
        with tenant_context(tenant_b):
            assert Unit.objects.filter(code='A-001').count() == 0, \
                "TENANT ISOLATION FAILED: Tenant B vê unidade do Tenant A"
    
    def test_cross_tenant_query_fails(self, tenant_a, tenant_b, project_tenant_a):
        """Tentar aceder dados de outro tenant diretamente deve falhar"""
        
        with tenant_context(tenant_a):
            project_a_id = project_tenant_a.id
        
        # Tentar aceder ao projeto do tenant A estando no tenant B
        with tenant_context(tenant_b):
            # Isto deve retornar vazio ou levantar exceção
            try:
                project = Project.objects.get(id=project_a_id)
                # Se chegou aqui, verificar se é realmente o projeto certo
                assert False, "TENANT ISOLATION FAILED: Query cross-tenant permitida"
            except Project.DoesNotExist:
                # Este é o comportamento esperado
                pass
    
    def test_simple_history_tenant_isolation(self, tenant_a, tenant_b, project_tenant_a, project_tenant_b):
        """Historico de alterações também deve estar isolado"""
        
        with tenant_context(tenant_a):
            project_tenant_a.name = 'Projeto A Atualizado'
            project_tenant_a.save()
            history_a_count = project_tenant_a.history.count()
        
        with tenant_context(tenant_b):
            project_tenant_b.name = 'Projeto B Atualizado'
            project_tenant_b.save()
            history_b_count = project_tenant_b.history.count()
        
        # Verificar que históricos não se misturam
        with tenant_context(tenant_a):
            for record in project_tenant_a.history.all():
                assert 'Tenant B' not in record.name, \

            for record in project_tenant_b.history.all():
                assert 'Tenant A' not in record.name, \
```

```python
import pytest
from rest_framework.test import APIClient
from django_tenants.utils import tenant_context

@pytest.mark.django_db
class TestAPIIsolation:
    """
    Testes de isolamento a nível de API
    """
    
    def test_api_user_cannot_access_other_tenant_data(self, tenant_a, tenant_b, project_tenant_a, user_tenant_b):
        """Utilizador autenticado não deve aceder dados de outro tenant via API"""
        
        client = APIClient()
        
        # Autenticar como user do Tenant B
        with tenant_context(tenant_b):
            login_response = client.post('/api/auth/login/', {
                'email': 'user@b.com',
                'password': 'password123',
            })
            token = login_response.data['access']
            client.credentials(HTTP_AUTHORIZATION=f'Bearer {token}')
        
        # Tentar aceder projeto do Tenant A
        response = client.get(f'/api/projects/{project_tenant_a.id}/')
        
        # Deve retornar 404 ou 403, nunca 200 com dados
        assert response.status_code in [404, 403], \
            f"API ISOLATION FAILED: Status {response.status_code}, dados expostos"
    
    def test_api_create_data_stays_in_tenant(self, tenant_a, user_tenant_a):
        """Dados criados via API devem ficar no tenant correto"""
        
        client = APIClient()
        
        with tenant_context(tenant_a):

                'email': 'user@a.com',
```

```bash
#!/bin/bash

set -e

# Wait for PostgreSQL
echo "Waiting for PostgreSQL..."
while ! nc -z $DB_HOST $DB_PORT; do
  sleep 0.1
done
echo "PostgreSQL started"

# Run migrations if needed
if [ "$RUN_MIGRATIONS" = "true" ]; then
  echo "Running migrations..."
  python manage.py migrate_schemas --shared
  python manage.py migrate_schemas
fi

# Collect static files
if [ "$COLLECT_STATIC" = "true" ]; then
  echo "Collecting static files..."
  python manage.py collectstatic --noinput
fi

exec "$@"
```

```makefile
	find . -type f -name "*.pyc" -delete
	find . -type d -name "__pycache__" -delete
	rm -rf .pytest_cache
	rm -rf htmlcov
	rm -rf coverage.xml

create-tenant:
	python manage.py create_tenant --name=$(name) --slug=$(slug) --domain=$(domain)

backup:
	pg_dump -h $(DB_HOST) -U $(DB_USER) $(DB_NAME) > backup_$(shell date +%Y%m%d).sql

	psql -h $(DB_HOST) -U $(DB_USER) $(DB_NAME) < $(file)
```
